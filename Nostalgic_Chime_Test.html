function chime(notes=[880,1318.5]){
  const ctx=new (window.AudioContext||window.webkitAudioContext)();
  const now=ctx.currentTime;
  notes.forEach((freq,i)=>{
    const o=ctx.createOscillator();
    const g=ctx.createGain();
    o.type='sine'; o.frequency.value=freq;
    o.connect(g); g.connect(ctx.destination);
    g.gain.setValueAtTime(0, now + i*0.22);
    g.gain.linearRampToValueAtTime(0.25, now + i*0.22 + 0.01);
    g.gain.exponentialRampToValueAtTime(0.0001, now + i*0.22 + 0.5);
    o.start(now + i*0.22); o.stop(now + i*0.22 + 0.52);
  });
}
